#include<iostream>
#include<Windows.h>
#include<string.h>
#include<fstream>
using namespace std;

int main() {
      SetConsoleCP(1251);
      SetConsoleOutputCP(1251);

      int n; //Кількість елементів массиву
      int i; //Параметр циклу
      int k, l;
      int max_l, min_l;
      string arr[10];
      string temp;

      cout << "Введіть кількість елементів масиву\n";
      cin >> n;
      for (i = 0; i <= n - 1; i)
      {
          cout << "Beдiть" << i + 1 << "-ий елемент массиву\t";   
          cin >> arr[i];
          cout << endl;
      }

      for (i = 0; i <= n - 1; i)
          cout << arr[i] << endl;

      max_l = size(arr[0]);
      min_l = size(arr[0]);

      k = 0;
      l = 0;

      for (i = 0; i <= n - 1; i)
      {
          if (max_l <= size(arr[i]))
          {
              max_l = size(arr[i]);
              k = i;
          }
          if (min_l >= size(arr[i]))
          {
              min_l = size(arr[i]);
              l = i;
          }
      }
      cout << endl;
      cout << "Максимальний елемент - ";
      cout << arr[k] << endl;
      cout << endl;
      cout << "Мінімальний елемент - ";
      cout << arr[l] << endl;
      cout << endl;
      cout << "Заміна елементів\n";
      temp = arr[l];
      arr[l] = arr[k];
      arr[k] = temp;
      for (i = 0; i <= n - 1; i)
          cout << arr[i] << endl;;

      cout << endl;;
      return 0;
    }
