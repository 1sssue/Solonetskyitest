      Реалізація лінійного пошуку

      #include iostream
      #include vector
      #include Windows.h

      using namespace std;

      int linearSearch(const vector'int'& arr, int target) {
      int steps = 0;
      for (int i = 0; i < arr.size(); ++i) {
      steps++;
      if (arr[i] == target) {
      cout << "Кількість виконаних дій: " << steps << endl;
      return i; // Повертаємо індекс знайденого елемента
      }
      }
      cout << "Кількість виконаних дій: " << steps << endl;
      return -1; // Якщо елемент не знайдено, повертаємо -1
      }

      int main() {
      SetConsoleCP(1251);
      SetConsoleOutputCP(1251);

      vector'int' arr = { 1, 3, 5, 7, 9, 11, 13, 15 };
      int target = 15;

      int linearResult = linearSearch(arr, target);

      if (linearResult != -1) {
      cout << "Елемент - " << target << "\nЗнайдений на " << linearResult + 1 << " позиції" << endl;
      }
      else {
      cout << "Елемент - " << target << " не знайдений" << endl;
      }
        return 0;
        }

      mov EAX, 2500
      imul X
      sub EAX, 8
      cdq
      div EDI
      shr EDI, 1
      cmp EDI, EDX
      adc EAX, 0
      mov dword ptr[EBX], EAX
      add EBX , 4

      add X, ESI
