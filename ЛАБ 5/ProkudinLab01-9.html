Програмування

#include 
 
int main()
{
    int x = 7;
    int y = 4;
 
    std::cout << "int / int = " << x / y << "\n";
    std::cout << "double / int = " << static_cast(x) / y << "\n";
    std::cout << "int / double = " << x / static_cast(y) << "\n";
    std::cout << "double / double = " << static_cast(x) / static_cast(y) << "\n";
 
    return 0;
}
