using namespace std;
                
int main()
    {
    SetConsoleCP(1251);
    SetConsoleOutputCP(1251);
                
    // Введення даних
    const int MaxN = 100;
    int arr[MaxN]{};
    int min;
    int n;
    int m = 0;
                
    // Введення кількості елементів масиву
    cout << "Введіть кількість елементів масиву" << endl;
    cin >> n;
                
    // Введення елементів масиву
    for (int i = 0; i <= n - 1; i++) {
        cout << "Введіть " << i + 1 << "-ий елемент масиву" << endl;
        cin >> arr[i];
        }

    min = arr[0];
                
    // Знаходження мінімального значення масиву
    for (int i = 0; i <= n - 1; i++) {
        if (min >= arr[i]) {
            min = arr[i];
            m = i;
            }
        }
                
    cout << "Мінімальне значення масиву " << min << endl;
                
    // Заміна мінімального числа масиву з останнім елементом масиву
    arr[m] = arr[n - 1];
    arr[n - 1] = min;
                
    // Вивід масиву
    for (int i = 0; i <= n - 1; i++) {
        cout << arr[i] << " ";
        }
        return 0;
    }
